public class Solution {
    public static void main(String[] args) {
        Solution solution = new Solution();

    public int climbStairs(int n) {
        if(n<=2) return n;
        // dp[i] ä»£è¡¨çš„æ˜¯åˆ°è¾¾å½“å‰æ¥¼æ¢¯çš„æ€»æ–¹æ³•æ•°
        int[] dp = new int[n];
        // åˆå§‹åŒ–
        dp[0] = 1;dp[1] = 2;

        for (int i = 2; i < n; i++) {
            dp[i] = dp[i-1]+dp[i-2];
        }
        return dp[n-1];
    }
}
