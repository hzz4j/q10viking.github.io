public class Solution {

    public static void main(String[] args) {
        Solution solution = new Solution();
        System.out.println(Arrays.toString(solution.searchRange(
                new int[]{5,7,7,8,8,10},
                8
        )));
    }
    public int[] searchRange(int[] nums, int target) {
        int[] res = new int[]{-1,-1};
        if(nums.length == 0){
            return res;
        }
        res[0] = findFirst(nums,target);
        // ç¬¬ä¸€ä¸ªéƒ½æ²¡æ‰¾åˆ°ï¼Œç›´æ¥ç»“æŸ
        if(res[0] == -1){
            return res;
        }
        res[1] = findLast(res[0],nums,target);
        return res;
    }

    private int findFirst(int[] nums,int target){
        int start = 0,end = nums.length - 1;
        int rec = -1;
        while(start <= end){
            int mid = (start + end) / 2;
            if(target <= nums[mid]){  // æ§åˆ¶æ–¹å‘,ä¼˜å…ˆå¾€å·¦ 5 7 7 8 8 10
                end = mid - 1;
            }else{
                start = mid + 1;
            }
            if(nums[mid] == target) {
                rec = mid;
            }
        }
        return rec;
    }

    private int findLast(int start,int[] nums,int target){
        int end = nums.length - 1;
        int rec = -1;
        while(start <= end){
            int mid = (start + end) / 2;
            if(target >= nums[mid]){  // æ§åˆ¶æ–¹å‘,ä¼˜å…ˆå¾€å³è¾¹ 5 7 7 8 8 10
                start = mid + 1;
            }else{
                end = mid - 1;
            }
            if(nums[mid] == target) {
                rec = mid;
            }
        }
        return rec;
    }
}
