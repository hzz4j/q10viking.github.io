public int coinChangeAll(int[] coins, int amount) {
        int n = coins.length;
        int impossible = amount + 1;
        /*dpæ•°ç»„ï¼Œå…¶ä¸­çš„å…ƒç´ dp[i][j] è¡¨ç¤ºä»å‰iç§ç¡¬å¸ä¸­é€‰ï¼Œ

        int[][] dp = new int[n + 1][impossible];

        for (int i = 0; i <= n; i++)

        dp[0][0] = 0;
        for (int i = 1; i <= n; i++) {
            int m = coins[i - 1];
            for (int j = 0; j <= amount; j++) {
                /*ä¸ç®¡ç¬¬iä¸ªç¡¬å¸å–å‡ ä¸ªï¼Œé‡‘é¢ä¸èƒ½è¶…è¿‡j*/
