public int[] twoSum(int[] nums, int target) {
    for (int i = 0; i < nums.length; i++) {
        for (int j = i+1; j < nums.length; j++) {
            if(target == nums[i] + nums[j])
                return new int[]{i,j};
        }
    }
    throw new IllegalArgumentException();
}

        Map<Integer,Integer> numToIndex = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            // æ‰¾åˆ°ç»“æœ
